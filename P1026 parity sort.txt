#include <iostream>
#include <queue>

using namespace std;

int main(){
    int n;
    cin>>n;
    int odd=0,even=0,oindex[n],eindex[n];
    priority_queue<int> odds,evens;
    for(int i=0;i<n;i++){
        int temp;
        cin>>temp;
        if(temp%2==0){ 
            evens.push(temp);
            eindex[even++]=i;
        }
        else{
            odds.push(temp);
            oindex[odd++]=i;
        }
    }
    int o=2147483647,e=2147483647;
    for(int i=n-1;i>=0;i--){
        if(eindex[even-1]==i){
            even--;
            e=evens.top();
            evens.pop();
            if(e>o){
                cout<<"NO";
                return 0;
            }
        }
        else{
            o=odds.top();
            odds.pop();
            if(o>e){
                cout<<"NO";
                return 0;
            }
        }
    }
    cout<<"YES";
    return 0;
}
